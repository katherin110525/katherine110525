Demonstrating JavaScript interactivity and dynamic styling

This is my activity in creating a simple Interactive Webpagefrom scratch using CodePen.io that demonstrates dynamic styling with JavaScript.</p> <button id='styleButton' onclick="changeStyle()">Change Style</button> <button onclick="resetStyle()">Reset Style</button> <button onclick="updateText()">Update Text</button> <button onclick="toggleVisibility()">Toggle Visibility</button> </body> </html>

CSS Code

CSS #demo { font-size: 40px; color: black; background-color: white; padding: 10px; border: 1px solid black; display: block; }

JavaScript Code

JavaScript let colors = ["green", "orange", "yellow"]; let currentIndex = 0; function changeStyle() { let para = document.getElementById("demo"); para.style.fontSize = "20px"; para.style.color = "white"; para.style.backgroundColor = colors[currentIndex]; currentIndex = (currentIndex + 1) % colors.length; } function resetStyle() { let para = document.getElementById("demo"); para.style.fontSize = "16px"; para.style.color = "var(--dark)"; para.style.backgroundColor = "white"; } function updateText() { document.getElementById("demo").innerText = "The text has been updated!"; } function toggleVisibility() { let para = document.getElementById("demo"); if (para.style.display === "none") { para.style.display = "block"; } else { para.style.display = "none"; } } document.getElementById("demo").addEventListener("click", function () { alert("You clicked the paragraph!"); });
